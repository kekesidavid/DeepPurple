#returns the path of an image by it's index number
def get_image_path_by_index(i):
    fp = open('../data/CUB_200_2011/images.txt', 'r')
    images_folder = '../data/CUB_200_2011/images/'
    cnt = 0
    while cnt < i:
        cnt += 1
        line = fp.readline()
    filename = line.split(' ')[1]
    return images_folder + filename.rstrip()

#returns the bounding box corners of an image's main content by it's index number
def get_bounding_boxes_by_index(i):
    fp = open('../data/CUB_200_2011/bounding_boxes.txt', 'r')
    cnt = 0
    while cnt < i:
        cnt += 1
        line = fp.readline()
    corners_string = line.rstrip().split(' ')
    corners_float = [float(numeric_string) for numeric_string in corners_string[1:5]]
    corners_int = [int(floats) for floats in corners_float]
    return corners_int
